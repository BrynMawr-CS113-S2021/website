class Mystery {
    String replaceChars(String initialString) {
        String newString = "";
        for (int i = 0; i < initialString.length(); i++) {
            char c = initialString.charAt(i);
            if (c == 's') {
                newString += "5";
            }
            else if (c == 'e') {
                newString += "3";
            }
            else {
                newString += c;
            }
        }
        return newString;
    }

    public static void main(String[] args) {
        String result1 = replaceChars("elephant");
        println(result1);

        String result2 = replaceChars("sour patch kids");
        println(result2);
    }
}

String reverse(String initialString) {
    String newString = "";
    for (int i = 0; i < initialString.length(); i++) {
        char c = initialString.charAt(i);
        newString = c + newString;
    }
    return newString;
}

int indexOfSmallest(int[] ls) {
   int smallest = 0;
   for (int i = 0; i < ls.length; i++) {
       if (ls[i] < ls[smallest]) {
           smallest = i;
       }
   }
   return smallest;
}
